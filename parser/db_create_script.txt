create database kl;
use kl;
CREATE TABLE IF NOT EXISTS `kl`.`Storage` (
    `link` CHAR(128) NOT NULL,
    `title` TEXT,
    `date` DATETIME,
    `section` TINYTEXT,
    `theme` TINYTEXT,
    `text` MEDIUMTEXT,
    `viewsCount` INT,
    `commentsCount` INT,
  PRIMARY KEY (`link`),
  UNIQUE INDEX `link_UNIQUE` (`link` ASC))
ENGINE = InnoDB;

CREATE TABLE IF NOT EXISTS `kl`.`splitted` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `link` CHAR(128) NOT NULL,
    `text` MEDIUMTEXT,
  PRIMARY KEY (`id`),
  UNIQUE INDEX `id_UNIQUE` (`id` ASC))
ENGINE = InnoDB;

CREATE TABLE IF NOT EXISTS `kl`.`persons` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `fullname` MEDIUMTEXT,
  PRIMARY KEY (`id`),
  UNIQUE INDEX `id_UNIQUE` (`id` ASC))
ENGINE = InnoDB;

CREATE TABLE IF NOT EXISTS `kl`.`places` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `name` MEDIUMTEXT,
  PRIMARY KEY (`id`),
  UNIQUE INDEX `id_UNIQUE` (`id` ASC))
ENGINE = InnoDB;

CREATE TABLE IF NOT EXISTS `kl`.`persons_synonyms` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `fullname` MEDIUMTEXT,
    `synonym` MEDIUMTEXT,
  PRIMARY KEY (`id`),
  UNIQUE INDEX `id_UNIQUE` (`id` ASC))
ENGINE = InnoDB;

CREATE TABLE IF NOT EXISTS `kl`.`places_synonyms` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `name` MEDIUMTEXT,
    `synonym` MEDIUMTEXT,
  PRIMARY KEY (`id`),
  UNIQUE INDEX `id_UNIQUE` (`id` ASC))
ENGINE = InnoDB;

CREATE TABLE IF NOT EXISTS `kl`.`filtered` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `link` CHAR(128) NOT NULL,
    `text` MEDIUMTEXT,
  PRIMARY KEY (`id`),
  UNIQUE INDEX `id_UNIQUE` (`id` ASC))
ENGINE = InnoDB;

CREATE TABLE IF NOT EXISTS `kl`.`tonality` (
    `id` INT NOT NULL AUTO_INCREMENT,
    `id_filtered` INT NOT NULL,
    `first_tonality` TEXT,
    `first_value` DOUBLE,
    `second_tonality` TEXT,
    `second_value` DOUBLE,
  PRIMARY KEY (`id`),
  UNIQUE INDEX `id_UNIQUE` (`id` ASC))
ENGINE = InnoDB;
